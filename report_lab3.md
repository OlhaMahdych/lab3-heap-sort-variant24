procedure heapSort(A)
    n := length(A)
    for i := n/2 - 1 downto 0 do
        Sink(A, i, n)

    for i := n - 1 downto 1 do
        swap(A[0], A[i])
        Sink(A, 0, i)

procedure Sink(A, i, n)
    largest := i
    left := 2*i + 1
    right := 2*i + 2

    if left < n and A[left] > A[largest] then
        largest := left
    if right < n and A[right] > A[largest] then
        largest := right

    if largest != i then
        swap(A[i], A[largest])
        Sink(A, largest, n)
